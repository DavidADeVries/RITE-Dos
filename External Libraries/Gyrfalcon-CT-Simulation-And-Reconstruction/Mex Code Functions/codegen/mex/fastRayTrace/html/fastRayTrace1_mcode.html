1function [waterEquivDoseSourceToIso, waterEquivDoseIsoToEpid] = fastRayTrace(pointSourceCoords, pointDetectorCoords, phantomLocationInM, phantomDims, voxelDimsInM, phantomData, sourceToAxisInM, axisToEpidInM)
  2% rawDetectorValue = fastRayTrace(pointSourceCoords, pointDetectorCoords, phantomLocationInM, phantomDims, voxelDimsInM, phantomData, startingIntensity)
  3
  4bounds1 = [phantomLocationInM(1), phantomLocationInM(2:3) - voxelDimsInM(2:3).*phantomDims(2:3)];
  5bounds2 = [phantomLocationInM(1)+voxelDimsInM(1)*phantomDims(1), phantomLocationInM(2:3)];
  6
  7[deltas, sourceStartingPoint] = createLineEquation(pointSourceCoords, pointDetectorCoords);
  8
  9invDeltas = 1 ./ deltas;
 10
 11tMinsTemp = (bounds1 - sourceStartingPoint) .* invDeltas;
 12tMaxsTemp = (bounds2 - sourceStartingPoint) .* invDeltas;
 13
 14tMins = tMinsTemp;
 15tMaxs = tMaxsTemp;
 16
 17negDeltas = deltas < 0;
 18
 19tMins(negDeltas) = tMaxsTemp(negDeltas);
 20tMaxs(negDeltas) = tMinsTemp(negDeltas);
 21
 22tMin = -Inf;
 23tMax = Inf;
 24
 25if deltas(1) ~= 0
 26    if tMins(1) > tMin
 27        tMin = tMins(1);
 28    end
 29    if tMaxs(1) < tMax
 30        tMax = tMaxs(1);
 31    end
 32elseif sourceStartingPoint(1) > bounds2(1) || sourceStartingPoint(1) < bounds1(1)
 33    tMin = Inf;
 34    tMax = -Inf;
 35end
 36
 37if deltas(2) ~= 0
 38    if tMins(2) > tMin
 39        tMin = tMins(2);
 40    end
 41    if tMaxs(2) < tMax
 42        tMax = tMaxs(2);
 43    end
 44elseif sourceStartingPoint(2) > bounds2(2) || sourceStartingPoint(2) < bounds1(2)
 45    tMin = Inf;
 46    tMax = -Inf;
 47end
 48
 49if deltas(3) ~= 0
 50    if tMins(3) > tMin
 51        tMin = tMins(3);
 52    end
 53    if tMaxs(3) < tMax
 54        tMax = tMaxs(3);
 55    end
 56elseif sourceStartingPoint(3) > bounds2(3) || sourceStartingPoint(3) < bounds1(3)
 57    tMin = Inf;
 58    tMax = -Inf;
 59end
 60
 61if tMax < tMin
 62    waterEquivDoseSourceToIso = 0;
 63    waterEquivDoseIsoToEpid = 0;
 64else % run through the voxels
 65    waterEquivDoseSourceToIso = 0;
 66    waterEquivDoseIsoToEpid = 0;
 67    
 68    % calc what t value represents being past isocentre
 69    sourceToDetectorInM = norm(pointSourceCoords - pointDetectorCoords);
 70    deltaNorm = norm(deltas);
 71    
 72    sinTheta = (sourceToAxisInM + axisToEpidInM) ./ sourceToDetectorInM;
 73    
 74    sourceToIsoForRayInM = sourceToAxisInM ./ sinTheta;
 75    
 76    tCrossover = sourceToIsoForRayInM ./ deltaNorm;
 77    
 78    sourceStartingPoint = sourceStartingPoint - phantomLocationInM; %shift over so corner is at origin
 79    
 80    currentT = tMin;
 81    currentPoint = sourceStartingPoint + currentT .* deltas;
 82    endingT = tMax;
 83        
 84    % have starting point and end point, now will find which voxels and with
 85    % what distances across each voxel the ray travels
 86           
 87    isVoxelDim0 = (voxelDimsInM == 0);
 88    
 89    isDelta0 = (deltas == 0);
 90    isDeltaNeg = (deltas < 0);
 91    invVoxelDims = 1 ./ voxelDimsInM;
 92    
 93    nextLatticeAdder = sign(deltas) .* [1 -1 -1];
 94    
 95    latticeToIndex = isDeltaNeg;
 96    latticeToIndex(1) = ~latticeToIndex(1);
 97    
 98    latticeToIndex(isDelta0) = 1; %needs to be, since we always floor to find the lattice for delta == 0
 99    
100    while endingT - currentT > Constants.Round_Off_Error_Bound
101        currentLattices = getLattices(currentPoint, invVoxelDims, isVoxelDim0, isDeltaNeg, isDelta0);
102        
103        nextLattices = currentLattices + nextLatticeAdder;
104        
105        tValsForNextLattices = ((nextLattices .* voxelDimsInM .* [1 -1 -1]) - sourceStartingPoint) .* invDeltas;
106                
107        nextT = min(tValsForNextLattices(~isDelta0));
108        
109        nextPoint = sourceStartingPoint + nextT .* deltas;
110        
111        length = norm(currentPoint - nextPoint);
112        indices = currentLattices + latticeToIndex;
113        
114        relativeElectronDensity = phantomData(indices(2), indices(1), indices(3));
115                
116        currentT = nextT;
117        currentPoint = nextPoint;
118        
119        if currentT <= tCrossover % in the source-to-iso area
120            waterEquivDoseSourceToIso = waterEquivDoseSourceToIso + length .* relativeElectronDensity;
121        else % in the iso-to-EPID area
122            waterEquivDoseIsoToEpid = waterEquivDoseIsoToEpid + length .* relativeElectronDensity;
123        end
124              
125    end
126    
127end
128
129end

133function lattices = getLattices(point, invVoxelDims, isVoxelDim0, isDeltaNeg, isDelta0)
134    
135    unroundedVals = point .* invVoxelDims;
136    
137    unroundedVals = [1 -1 -1].*unroundedVals;
138            
139    % first need to kill off any rounding errors
140    unroundedVals = roundToLevel(unroundedVals, Constants.Round_Off_Level);
141    
142    % no round to get lattice/index values
143    isDeltaNeg(1) = ~isDeltaNeg(1);
144    selectFloor = isDeltaNeg | isDelta0; % ceiling if delta is positive ONLY
145        
146    floorVals = floor(unroundedVals(selectFloor));
147    ceilVals = ceil(unroundedVals(~selectFloor));
148     
149    lattices =  zeros(1,3);
150    
151    lattices(selectFloor) = floorVals;
152    lattices(~selectFloor) = ceilVals;
153    lattices(isVoxelDim0) = 0;
154end
155
156function roundedValues = roundToLevel(unroundedValues, level)
157    roundedValues = floor((unroundedValues.*(10^level)) + 0.5).*(10^(-level));
158end
